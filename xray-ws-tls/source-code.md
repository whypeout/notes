    listen xxx.xxx.xxx.xxx:80;
    server_name www.example.net;

    rewrite ^(.*) https://$server_name$1 permanent;

    listen xxx.xxx.xxx.xxx:443 ssl http2;
    server_name www.example.net;

    root   /vhosts/www.example.net;
